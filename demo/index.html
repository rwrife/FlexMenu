Integrating with Frameworks

FlexMenu is pure Javascript, making it simple to instantiate in any modern library. Here's how:

import { FlexMenu } from 'flexmenu';
import 'flexmenu/style.css';

const menu = new FlexMenu('#nav', {
  sticky: true,
  onOverflow: (item) => console.log('Overflowed:', item),
  onUnderflow: (item) => console.log('Returned:', item)
});
import React, { useEffect, useRef } from 'react';
import { FlexMenu } from 'flexmenu';
import 'flexmenu/style.css';

export function Navigation() {
  const navRef = useRef(null);
  const flexMenuInstance = useRef(null);

  useEffect(() => {
    if (navRef.current) {
      flexMenuInstance.current = new FlexMenu(navRef.current, {
        sticky: false
      });
    }

    return () => {
      // Clean up event listeners & observers on unmount!
      flexMenuInstance.current?.destroy();
    };
  }, []);

  return (
    <ul id="nav" ref={navRef} className="flexmenu-nav">
      <li><a href="#">Home</a></li>
      <li><a href="#">Products</a></li>
      <li><a href="#">Services</a></li>
    </ul>
  );
}
<template>
  <ul id="nav" ref="navElement" class="flexmenu-nav">
    <li><a href="#">Home</a></li>
    <li><a href="#">Blog</a></li>
    <li><a href="#">Contact</a></li>
  </ul>
</template>

<script setup>
import { ref, onMounted, onUnmounted } from 'vue';
import { FlexMenu } from 'flexmenu';
import 'flexmenu/style.css';

const navElement = ref(null);
let flexMenu = null;

onMounted(() => {
  if (navElement.value) {
    flexMenu = new FlexMenu(navElement.value);
  }
});

onUnmounted(() => {
  if (flexMenu) {
    flexMenu.destroy();
  }
});
</script>
import { Component, ElementRef, OnInit, OnDestroy, ViewChild } from '@angular/core';
import { FlexMenu } from 'flexmenu';

@Component({
  selector: 'app-navigation',
  template: `
    <ul #nav class="flexmenu-nav">
      <li><a href="#">Home</a></li>
      <li><a href="#">About</a></li>
    </ul>
  `,
  styleUrls: ['./navigation.component.css']
})
export class NavigationComponent implements OnInit, OnDestroy {
  @ViewChild('nav', { static: true }) navElement!: ElementRef;
  private flexMenu?: FlexMenu;

  ngOnInit() {
    this.flexMenu = new FlexMenu(this.navElement.nativeElement);
  }

  ngOnDestroy() {
    this.flexMenu?.destroy();
  }
}
